21  @ActiveProfiles({"default"})
22  @RunWith(SpringRunner.class)
23  @SpringBootTest(properties = {
24      "debug=false", // debug=false can not disable AUTO-CONFIGURATION REPORT
25      "logging.level.root=INFO"
26  })
27  @Import(LoggingTestApplication.class)
28  @Slf4j
29  public class SpringBootLog4j2LoggingLevelTest {
30  
31    private static final PrintStream ORIGINAL_SYS_OUT = System.out;
32    private static final PrintStream ORIGINAL_SYS_ERR = System.err;
33  
34    private static final ByteArrayOutputStream OUT = new ByteArrayOutputStream();
35    private static final ByteArrayOutputStream ERR = new ByteArrayOutputStream();
36  
37    @BeforeClass
38    public static void setUpStreams() {
39      // SystemOutRule can get logback's console output but can not get log4j2's console output
40      System.setOut(new PrintStream(OUT));
41      System.setErr(new PrintStream(ERR));
42    }
43  
44    @AfterClass
45    public static void restoreStreams() {
46      System.setOut(ORIGINAL_SYS_OUT);
47      System.setErr(ORIGINAL_SYS_ERR);
48    }
49  
50    @Autowired
51    private LoggingTestApplication application;
52  
53    @Test
54    public void testLevel() throws InterruptedException {
55      this.application.infoLog("hello, {}", "world");
56      ORIGINAL_SYS_OUT.println(OUT.toString());
57      assertTrue(OUT.toString().contains("hello, world"));
58  
59      this.application.debugLog("hello, {}", "debug log");
60      assertFalse(OUT.toString().contains("hello, debug log"));
61    }
62  }
